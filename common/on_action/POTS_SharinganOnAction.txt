on_birthday = {
    on_actions = {
        sharingan_birthday_one
        sharingan_birthday_two
        sharingan_birthday_three
    }
}

sharingan_birthday_one = {
    trigger = {
        AND = {
            age >= 18
            has_trait = uchihagene
            is_ai = yes
        }
    }

    events = {
        sharingan_level.1
    }
}

sharingan_birthday_two = {
    trigger = {
        AND = {
            age >= 20
            has_trait = onetomoe
            is_ai = yes
        }
    }

    events = {
        sharingan_level.2
    }
}

sharingan_birthday_three = {
    trigger = {
        AND = {
            age >= 22
            has_trait = twotomoe
            is_ai = yes
        }
    }

    events = {
        sharingan_level.3
    }
}
