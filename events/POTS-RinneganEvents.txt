RinneganEvents.001 = {
    type = character_event
    title = RinneganEvents.001.t
    desc = RinneganEvents.001.d
    theme = martial

    right_portrait = {
        character = scope:actor
        animation = worry
    }

    option = {
        name = RinneganEvents.001.a

            }
        }
    }

    option = {
        name = RinneganEvents.001.b
        trigger = {
            has_trait = eternalsharingan
            NOT = { has_trait = rinnegan}
        }
        hidden_effect = {
            add_gold = 100
        }
    }
}

    type = character_event
    title = RinneganEvents.003.t
    desc = RinneganEvents.003.d

        character = scope:actor
        animation = worry
    }
    left_portrait = {
        character = scope:recipient
        animation = worry
    }

    option = {
        name = RinneganEvents.003.a
        trigger = {
            has_trait = rinnegan

            scope:recipient = {
                remove_trait = one_tail

            }
            scope:actor = {
                add_character_modifier = {
                    modifier = summoned_gedo_mazo_modifier
                    months = 2
                }
            }
        }
    }
    option = {
        name = RinneganEvents.003.b
        trigger = {
            has_trait = rinnegan
            has_trait = one_tail
            NOT = { has_trait = two_tail }
        }
        add_trait = two_tail
        hidden_effect = {
            scope:recipient = {
                remove_trait = two_tail
                death = {
                    killer = scope:actor
                    death_reason = tailedbeast_absorbed
                }
            }
            scope:actor = {
                add_character_modifier = {
                    modifier = summoned_gedo_mazo_modifier

                }
            }
